import sys
import os
import telnetlib
import time
#

Constructor for a telnet library to connect to Guess See Guess - http://graphexploration.cond.org/

class Dnet():
# 
    def __init__(self, gr):
        self.graph=gr
        if self.graph:
            self.tn=telnetlib.Telnet('localhost', 2222)
#

Send a message to Guess via telnet. Guess - http://graphexploration.cond.org/

    def send(self, cmd):
# 
        if self.graph:
            self.tn.write(cmd+"\n")
            self.tn.read_until("Done")
