def compareFiles(x, y):
    assert isinstance(x, str)
    assert isinstance(y, str)
    x = x.replace(path, '').split('__')
    y = y.replace(path, '').split('__')
    fmt = '%Y-%m-%d_%Hh%Mm%Ss'
    a = time.strptime(x[0], fmt)
    b = time.strptime(y[0], fmt)
    if a > b:
        return 1
    elif a < b:
        return -1
    else:
        return 0
